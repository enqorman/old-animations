// The below code would be applied after the if block for the darkSkyBuffer (d < 0.0f)
// But before the last 2 calls for setShaderColor & depthMask

if (this.world.getDimensionEffects().isAlternateSkyColor()) {
	RenderSystem.setShaderColor(f * 0.2F + 0.04F, g * 0.2F + 0.04F, h * 0.6F + 0.1F, 1.0F);
} else {
	RenderSystem.setShaderColor(f, g, h, 1.0F);
}

// If Statement is a fix for it showing below y -64 (It didn't use to show below y 0/-64)
double d = this.client.player.getCameraPosVec(f).y - this.world.getLevelProperties().getSkyDarknessHeight(this.world);
if (d > 0.0) {
	matrixStack.push();
	matrixStack.translate(0.0F, -((float) (d - 16.0)), 0.0F);
	this.darkSkyBuffer.bind();
	this.darkSkyBuffer.draw(matrixStack.peek().getPositionMatrix(), matrix4f2, shaderProgramLocalRef.get());
	VertexBuffer.unbind();
	matrixStack.pop();
}
